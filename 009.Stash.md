# Stash

# Pasos para agregar los cambios a un stash

git add .
git stash

# Listar los stash

git stash list

# Recuperar un stash

git stash pop

# Borrar u stash

Elminar el stash mas reciente
git stash drop

Eliminar un stash en base a un nombre
git stash drop <nombre>

Elminar todos los stash
git stash clear

# Aplicar un stash

git stash apply stash@{0}

# Ver un stash
git stash show <nombre>
